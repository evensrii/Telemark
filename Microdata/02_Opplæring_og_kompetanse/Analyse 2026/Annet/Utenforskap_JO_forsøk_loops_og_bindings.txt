require no.ssb.fdb:51 as fdb 
create-dataset pop

//Sett startår 
let år0 = 2019
let antall_år = 5
let siste_år = $år0 + $antall_år

// ===== ÅR 0 (Basisår) - Populasjonsoppsett =====

let dato_år0 = date_fmt($år0, 11, 16)
let dato2_år0 = date_fmt($år0, 11, 30)

//henter Prioritert arbeidsstyrkestatus
import fdb/ARBSTATUS_PERS_UTENFOR $dato_år0 as ARBSTATUS ++ $år0
tabulate ARBSTATUS ++ $år0, missing

// Forenkle Status
generate Status ++ $år0 = "5 - Ukjent"
replace Status ++ $år0 = "1 - Innenfor" if ARBSTATUS ++ $år0 == '0'
replace Status ++ $år0 = "2 - Utenfor" if ARBSTATUS ++ $år0 == '1'

tabulate Status ++ $år0

//legger på geografi 
import fdb/ARBSTATUS_PERS_KOMM_NR $dato2_år0  as bosted ++ $år0
generate fylke = substr(bosted ++ $år0,1,2)
generate kommnr = substr(bosted ++ $år0,1,4)
keep if fylke == '50'| fylke == '16'| fylke  == '17'  | kommnr  == '1567'| kommnr  == '1571'

tabulate bosted ++ $år0 Status ++ $år0, missing

//legger til info om kjønn
import fdb/BEFOLKNING_KJOENN as kjønn

//setter aldersintervall i første år
import fdb/ARBSTATUS_PERS_ALDER $dato_år0 as alder ++ $år0

keep if alder ++ $år0 >= 15
keep if alder ++ $år0 <= 29

tabulate alder ++ $år0 Status ++ $år0, missing rowpct freq
tabulate alder ++ $år0 Status ++ $år0
tabulate kjønn Status ++ $år0, freq colpct rowpct
tabulate bosted ++ $år0 Status ++ $år0, missing

// ===== ÅR 1-4 (Oppfølgingsår) =====

for i in 1 : $antall_år
  let gjeldende_år = $år0 + $i
  let neste_år = $gjeldende_år + 1
  let dato_i = date_fmt($gjeldende_år, 11, 16)
  let dato1_i = date_fmt($neste_år, 1, 1)

  // Import arbeidsstyrkestatus
  import fdb/ARBSTATUS_PERS_UTENFOR $dato_i as ARBSTATUS ++ $gjeldende_år

  //skjekk på statuskode for ukjent - merk 1 &1/2  månders forskjelle på telledato
  import fdb/BEFOLKNING_STATUSKODE $dato1_i as statuskode ++ $gjeldende_år

  // Generer Status-variabel
  generate Status ++ $gjeldende_år = "5 - Ukjent"
  replace Status ++ $gjeldende_år = "1 - Innenfor" if ARBSTATUS ++ $gjeldende_år == '0'
  replace Status ++ $gjeldende_år = "2 - Utenfor" if ARBSTATUS ++ $gjeldende_år == '1'
  replace Status ++ $gjeldende_år = "3 - Utvandret" if statuskode ++ $gjeldende_år == '3'
  replace Status ++ $gjeldende_år = "4 - Død" if statuskode ++ $gjeldende_år == '5'
end

// Valgfri diagnostikk - fjern kommentar for å kjøre
//tabulate ARBSTATUS ++ $år0 + 1 Status ++ $år0, missing
//tabulate ARBSTATUS ++ $år0 + 1 statuskode ++ $år0 + 1, missing
//tabulate ARBSTATUS ++ $år0 + 2 Status ++ $år0, missing
//tabulate ARBSTATUS ++ $år0 + 2 statuskode ++ $år0 + 2, missing
//tabulate ARBSTATUS ++ $år0 + 3 Status ++ $år0, missing
//tabulate ARBSTATUS ++ $år0 + 3 statuskode ++ $år0 + 3, missing
//tabulate ARBSTATUS ++ $år0 + 4 Status ++ $år0, missing
//tabulate ARBSTATUS ++ $år0 + 4 statuskode ++ $år0 + 4, missing
//tabulate ARBSTATUS ++ $år0 + 5 Status ++ $år0, missing
//tabulate ARBSTATUS ++ $år0 + 5 statuskode ++ $år0 + 5, missing

// ===== Tabulering og Sankey =====

// År 0 vs År 1
tabulate Status ++ $år0 Status ++ $år0 + 1, missing

// År 0 -> År 1 -> År 2
sankey Status ++ $år0 Status ++ $år0 + 1 Status ++ $år0 + 2
tabulate Status ++ $år0 Status ++ $år0 + 1 Status ++ $år0 + 2, missing

// År 0 -> År 1 -> År 2 -> År 3
sankey Status ++ $år0 Status ++ $år0 + 1 Status ++ $år0 + 2 Status ++ $år0 + 3
tabulate Status ++ $år0 Status ++ $år0 + 3, missing

// År 0 -> År 1 -> År 2 -> År 3 -> År 4
sankey Status ++ $år0 Status ++ $år0 + 1 Status ++ $år0 + 2 Status ++ $år0 + 3 Status ++ $år0 + 4
tabulate Status ++ $år0 + 3 Status ++ $år0 + 4, missing

tabulate alder ++ $år0 Status ++ $siste_år, missing rowpct freq
tabulate alder ++ $år0 Status ++ $siste_år

// ===== Vedvarende utenfor ===== (Alle som er utenfor alle årene)

generate vedvarende = "1 - vedvarende utenfor"
replace vedvarende = "2 - ikke vedvarende utenfor" if Status ++ $år0 == "1 - Innenfor"

for i in 1 : $antall_år
  let gjeldende_år = $år0 + $i
  replace vedvarende = "2 - ikke vedvarende utenfor" if Status ++ $gjeldende_år != '2 - Utenfor'
end

//tester
for i in 0 : $antall_år
  let gjeldende_år = $år0 + $i
  tabulate vedvarende Status ++ $gjeldende_år
end

tabulate kjønn vedvarende Status ++ $år0, missing
tabulate kjønn vedvarende if Status ++ $år0 == '2 - Utenfor', missing freq colpct rowpct

tabulate bosted ++ $år0 vedvarende, missing
tabulate bosted ++ $år0 Status ++ $år0, missing

tabulate alder ++ $år0 vedvarende, missing rowpct freq

// ===== Vedvarende innenfor ===== (Alle som er innenfor alle årene)

generate vedvarende_I = "1 - vedvarende innenfor"
replace vedvarende_I = "2 - ikke vedvarende innenfor" if Status ++ $år0 == "2 - Utenfor"

for i in 1 : $antall_år
  let gjeldende_år = $år0 + $i
  replace vedvarende_I = "2 - ikke vedvarende innenfor" if Status ++ $gjeldende_år != "1 - Innenfor"
end

//tester
for i in 0 : $antall_år
  let gjeldende_år = $år0 + $i
  tabulate vedvarende_I Status ++ $gjeldende_år
end

tabulate kjønn vedvarende_I Status ++ $år0, missing
tabulate kjønn vedvarende_I if Status ++ $år0 == '1 - Innenfor', missing freq colpct rowpct

tabulate vedvarende_I vedvarende Status ++ $siste_år, missing

// ===== Opptelling antall år utenfor =====

for i in 0 : $antall_år
  let gjeldende_år = $år0 + $i
  generate count ++ $gjeldende_år = 0
  replace count ++ $gjeldende_år = 1 if Status ++ $gjeldende_år == "2 - Utenfor"
end

generate antall_utenfor = (count ++ $år0) + (count ++ $år0 + 1) + (count ++ $år0 + 2) + (count ++ $år0 + 3) + (count ++ $år0 + 4) + (count ++ $år0 + 5)

//test på om opptelling ble rett
tabulate antall_utenfor vedvarende, missing

tabulate alder ++ $år0 antall_utenfor

// ===== Analyse =====

//utdanningsnivå
let dato_ut = date_fmt($siste_år, 8, 1)

import fdb/NUDB_BU $dato_ut as utd
generate utd1 = substr(utd, 1, 1)

generate utdnivå ++ $siste_år = '9'
replace utdnivå ++ $siste_år = '1' if utd1 == '1' | utd1 == '2'
replace utdnivå ++ $siste_år = '2' if utd1 == '3' | utd1 == '4' | utd1 == '5'
replace utdnivå ++ $siste_år = '3' if utd1 == '6'
replace utdnivå ++ $siste_år = '4' if utd1 == '7' | utd1 == '8'
define-labels utdnivtxt '1' 'Grunnskolenivå' '2' 'Videregående grunnopplæring' '3' 'Universitets og høgskolenivå kort' '4' 'Universitets og høgskolenivå lang' '9' 'Uoppgitt el ingen utdanning'
assign-labels utdnivå ++ $siste_år utdnivtxt

tabulate utdnivå ++ $siste_år Status ++ $siste_år, missing rowpct freq
tabulate utdnivå ++ $siste_år Status ++ $siste_år, missing colpct freq

tabulate utdnivå ++ $siste_år vedvarende, missing rowpct freq
tabulate utdnivå ++ $siste_år vedvarende, missing colpct freq

tabulate utdnivå ++ $siste_år vedvarende if Status ++ $år0 == '2 - Utenfor', missing rowpct freq
tabulate utdnivå ++ $siste_år vedvarende if Status ++ $år0 == '2 - Utenfor', missing colpct freq

tabulate utdnivå ++ $siste_år Status ++ $år0, missing
tabulate utdnivå ++ $siste_år vedvarende if Status ++ $år0 == '2 - Utenfor', missing
tabulate utdnivå ++ $siste_år antall_utenfor, missing

// uføre
let dato_ufør = date_fmt($siste_år, 12, 1)
import fdb/UFOERP2011FDT_GRAD $dato_ufør as Ufør ++ $siste_år

tabulate Ufør ++ $siste_år vedvarende, missing freq rowpct colpct
tabulate antall_utenfor Ufør ++ $siste_år, missing

//Innvandrere
import fdb/BEFOLKNING_INVKAT as kat
generate Innvandringskategori = '3'
replace Innvandringskategori = '1' if kat == 'B'
replace Innvandringskategori = '2' if kat == 'C'
tabulate Innvandringskategori, missing
//replace Innvandringskategori = 'Ukjent ' if sysmiss(kat) - "Stoppet av avsløringskontroll. For mange tabellceller med lav frekvens" --> Da er det ikke et problem uansett.
define-labels Innvandringskategoritxt '1' 'Innvandrere' '2' 'Norskfødt med innvandrerforeldre' '3' 'Øvrig befolkning'
assign-labels Innvandringskategori Innvandringskategoritxt

tabulate Innvandringskategori Status ++ $år0, missing rowpct colpct freq
tabulate Innvandringskategori Status ++ $siste_år, missing rowpct colpct freq
tabulate Innvandringskategori Status ++ $siste_år if Status ++ $år0 == '2 - Utenfor', missing rowpct colpct freq
tabulate Innvandringskategori vedvarende, missing rowpct freq
tabulate Innvandringskategori vedvarende if Status ++ $år0 == '2 - Utenfor', missing rowpct freq
tabulate antall_utenfor Innvandringskategori, missing

//Sosial bakgrunn
import fdb/NUDB_SOSBAK as Sosial_bakgrunn

tabulate Sosial_bakgrunn Innvandringskategori, missing
tabulate Sosial_bakgrunn Status ++ $siste_år, missing rowpct freq
tabulate Sosial_bakgrunn vedvarende if Status ++ $år0 == '2 - Utenfor', missing rowpct freq
tabulate antall_utenfor Sosial_bakgrunn, missing

// ===== Detaljert status (Nivå2) for år 0 og siste år =====

for yr in $år0 $siste_år
  let dato_yr = date_fmt($yr, 11, 16)
  import fdb/ARBSTATUS_PERS_STATUS8 $dato_yr as detaljert ++ $yr
  generate status4 ++ $yr = substr(detaljert ++ $yr, 1, 4)
  tabulate status4 ++ $yr, missing

  generate Nivå2_ ++ $yr = '0'
  replace Nivå2_ ++ $yr = '1 - Sysselsatte' if status4 ++ $yr == '0808' | status4 ++ $yr == '0816' | status4 ++ $yr == '0824' | status4 ++ $yr == '0832' | status4 ++ $yr == '1608'
  replace Nivå2_ ++ $yr = '4 - Registrerte arbeidsledige' if status4 ++ $yr == '2408'
  replace Nivå2_ ++ $yr = '3 - Deltakere på arbeids-markedstiltak' if status4 ++ $yr == '3208' | status4 ++ $yr == '3216' | status4 ++ $yr == '3232' | status4 ++ $yr == '4816' | status4 ++ $yr == '3240'
  replace Nivå2_ ++ $yr = '2 - Under ordinær utdanning' if status4 ++ $yr == '4008'
  replace Nivå2_ ++ $yr = '5 - Mottakere av arbeidsavklarings-penger/uføretrygd' if status4 ++ $yr == '4848' | status4 ++ $yr == '4832' | status4 ++ $yr == '4856' | status4 ++ $yr == '4864'
  replace Nivå2_ ++ $yr = '6 - Mottakere av AFP/alderspensjon' if status4 ++ $yr == '5616' | status4 ++ $yr == '5632'
  replace Nivå2_ ++ $yr = '7 - Andre' if status4 ++ $yr == '3224' | status4 ++ $yr == '4808' | status4 ++ $yr == '5608' | status4 ++ $yr == '5624' | status4 ++ $yr == '5640' | status4 ++ $yr == '6408' | status4 ++ $yr == '6416' | status4 ++ $yr == '6424' | status4 ++ $yr == '6432'
  replace Nivå2_ ++ $yr = '8 - Ukjent' if status4 ++ $yr == '7208'
end

tabulate Nivå2_ ++ $år0 Status ++ $år0
tabulate antall_utenfor Nivå2_ ++ $år0

tabulate Nivå2_ ++ $siste_år Status ++ $år0

//legger på død og utvandret for siste år
replace Nivå2_ ++ $siste_år = "0 - Utvandret" if statuskode ++ $siste_år == '3'
replace Nivå2_ ++ $siste_år = "9 - Død" if statuskode ++ $siste_år == '5'

tabulate Nivå2_ ++ $siste_år Status ++ $år0
tabulate Nivå2_ ++ $siste_år Nivå2_ ++ $år0

tabulate Nivå2_ ++ $siste_år vedvarende if Status ++ $år0 == '2 - Utenfor', missing rowpct freq

tabulate Nivå2_ ++ $siste_år Innvandringskategori if vedvarende == '1 - vedvarende utenfor', missing rowpct colpct freq
tabulate Nivå2_ ++ $siste_år Innvandringskategori if vedvarende == '1 - vedvarende utenfor', missing

tabulate Nivå2_ ++ $siste_år Nivå2_ ++ $år0 if vedvarende == '1 - vedvarende utenfor', missing

tabulate detaljert ++ $år0 vedvarende, missing
tabulate detaljert ++ $siste_år vedvarende, missing

tabulate detaljert ++ $år0 Status ++ $siste_år, missing
tabulate detaljert ++ $år0 Status ++ $år0, missing
tabulate detaljert ++ $siste_år Status ++ $siste_år, missing
tabulate detaljert ++ $siste_år Status ++ $år0, missing
